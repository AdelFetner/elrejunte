\subsection{Compilar C++20 con g++}
\begin{code}
g++ -std=c++20 {file} -o {filename}
Para Geany:
compile: g++ -DANARAP -std=c++20 -g -O2 -Wconversion -Wshadow -Wall -Wextra -c \"%f\"
build:   g++ -DANARAP -std=c++20 -g -O2 -Wconversion -Wshadow -Wall -Wextra -o \"%e\" \"%f\"
\end{code}
